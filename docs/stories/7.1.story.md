```typescript
export async function getUserSkills(userId: string): Promise<Skill[]> {
  return db
    .select()
    .from(skills)
    .where(and(eq(skills.userId, userId), isNull(skills.archivedAt)))
    .orderBy(asc(skills.name))
}
```
